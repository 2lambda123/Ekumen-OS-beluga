58 template <class Range, class Scalar>
59 Eigen::Matrix2<Scalar> covariance(Range&& range, const Eigen::Vector2<Scalar>& mean) {
60  auto view = range | ranges::views::common;
61  Eigen::Vector3<Scalar> coefficients = std::transform_reduce(
62  view.begin(), view.end(), Eigen::Vector3<Scalar>::Zero().eval(), std::plus{}, [mean](const auto& value) {
63  const auto centered = value - mean;
64  return Eigen::Vector3<Scalar>{
65  centered.x() * centered.x(),
66  centered.x() * centered.y(),
67  centered.y() * centered.y(),
68  };
69  });
70  coefficients /= (static_cast<Scalar>(view.size() - 1));
71  auto covariance_matrix = Eigen::Matrix2<Scalar>{};
72  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
73  return covariance_matrix;
74 }

91 template <
92  class Range,
93  class Pose = ranges::range_value_t<Range>,
94  class Scalar = typename Pose::Scalar,
95  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE2<Scalar>>>>
96 std::pair<Sophus::SE2<Scalar>, Eigen::Matrix3<Scalar>> estimate(Range&& range) {
97  auto pose_view = range | ranges::views::common;
98  auto translation_view = range | ranges::views::transform([](const auto& pose) { return pose.translation(); });
99 
100  // Compute the average of all the coefficients of the SE(2) group elements and
101  // construct a new SE(2) element.
102  const Eigen::Vector4<Scalar> sum_vector = std::transform_reduce(
103  pose_view.begin(), pose_view.end(), Eigen::Vector4<Scalar>::Zero().eval(), std::plus{},
104  [](const auto& pose) { return Eigen::Map<const Eigen::Vector4<Scalar>>{pose.data()}; });
105  const Eigen::Vector4<Scalar> mean_vector = sum_vector / static_cast<double>(pose_view.size());
106  Sophus::SE2<Scalar> mean = Eigen::Map<const Sophus::SE2<Scalar>>{mean_vector.data()};
107 
108  // Compute the covariance of the translation part.
109  Eigen::Matrix3<Scalar> covariance_matrix = Eigen::Matrix3<Scalar>::Zero();
110  covariance_matrix.template topLeftCorner<2, 2>() = covariance(translation_view, mean.translation());
111 
112  // Compute the circular variance and re-normalize the rotation component.
113  if (mean.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
114  // Handle the case where both averages are too close to zero.
115  // Return zero yaw and infinite variance.
116  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
117  mean.so2() = Sophus::SO2<Scalar>{0.0};
118  } else {
119  // See circular standard deviation in
120  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
121  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(mean.so2().unit_complex().norm());
122  mean.so2().normalize();
123  }
124  return std::pair{mean, covariance_matrix};
125 }
