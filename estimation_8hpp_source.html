43 template <class Range, class Scalar>
44 Eigen::Matrix2<Scalar> covariance(const Range& range, const Eigen::Vector2<Scalar>& mean) {
45  Eigen::Vector3<Scalar> coefficients = std::transform_reduce(
46  ranges::begin(range), ranges::end(range), Eigen::Vector3<Scalar>::Zero().eval(), std::plus{},
47  [mean](const auto& value) {
48  const auto centered = value - mean;
49  return Eigen::Vector3<Scalar>{
50  centered.x() * centered.x(),
51  centered.x() * centered.y(),
52  centered.y() * centered.y(),
53  };
54  });
55  coefficients /= (static_cast<Scalar>(ranges::size(range) - 1));
56  auto covariance_matrix = Eigen::Matrix2<Scalar>{};
57  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
58  return covariance_matrix;
59 }

94  [[nodiscard]] std::pair<Sophus::SE2d, Eigen::Matrix3d> estimated_pose() const {
95  const auto poses_view = this->self().states() | ranges::views::common;
96  const auto translation_view =
97  poses_view | ranges::views::transform([](const auto& pose) { return pose.translation(); });
98 
99  // Compute the average of all the coefficients of the SE(2) group elements and
100  // construct a new SE(2) element.
101  const Eigen::Vector4d sum_vector = std::transform_reduce(
102  poses_view.begin(), poses_view.end(), Eigen::Vector4d::Zero().eval(), std::plus{},
103  [](const auto& pose) { return Eigen::Map<const Eigen::Vector4d>{pose.data()}; });
104  const Eigen::Vector4d mean_vector = sum_vector / static_cast<double>(poses_view.size());
105  Sophus::SE2d mean = Eigen::Map<const Sophus::SE2d>{mean_vector.data()};
106 
107  // Compute the covariance of the translation part.
108  Eigen::Matrix3d covariance_matrix = Eigen::Matrix3d::Zero();
109  covariance_matrix.topLeftCorner<2, 2>() = covariance(translation_view, mean.translation());
110 
111  // Compute the circular variance and re-normalize the rotation component.
112  if (mean.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
113  // Handle the case where both averages are too close to zero.
114  // Return zero yaw and infinite variance.
115  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
116  mean.so2() = Sophus::SO2d{0.0};
117  } else {
118  // See circular standard deviation in
119  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
120  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(mean.so2().unit_complex().norm());
121  mean.so2().normalize();
122  }
123  return std::pair{mean, covariance_matrix};
124  }
125 };
