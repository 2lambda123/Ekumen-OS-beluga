64 template <class Range, class WeightsRange, class Scalar>
65 Eigen::Matrix2<Scalar>
66 calculate_covariance(Range&& range, WeightsRange&& normalized_weights, const Eigen::Vector2<Scalar>& mean) {
67  auto translations_view = range | ranges::views::common;
68  auto normalized_weights_view = normalized_weights | ranges::views::common;
69 
70  auto calculate_weighted_sample_covariance = [mean](const auto& value, const auto& weight) {
71  const auto centered = value - mean;
72  return Eigen::Vector3<Scalar>{
73  weight * centered.x() * centered.x(), // weighted sample x autocovariance
74  weight * centered.x() * centered.y(), // weighted sample xy cross-covariance
75  weight * centered.y() * centered.y(), // weighted sample y autocovariance
76  };
77  };
78 
79  // calculate the averaging factor for the weighted covariance estimate
80  // See https://en.wikipedia.org/wiki/Sample_mean_and_covariance#Weighted_samples
81  const auto squared_weight_sum = std::transform_reduce(
82  normalized_weights_view.begin(), normalized_weights_view.end(), Scalar{0.0}, std::plus{},
83  [](const auto& weight) { return (weight * weight); });
84 
85  // calculate the x autocovariance, xy cross-covariance, and y autocovariance
86  Eigen::Vector3<Scalar> coefficients =
87  std::transform_reduce(
88  translations_view.begin(), translations_view.end(), normalized_weights_view.begin(),
89  Eigen::Vector3<Scalar>::Zero().eval(), std::plus{}, calculate_weighted_sample_covariance) /
90  (1.0 - squared_weight_sum);
91 
92  // create the symmetric 2x2 translation covariance matrix from the coefficients
93  auto covariance_matrix = Eigen::Matrix2<Scalar>{};
94  covariance_matrix << coefficients(0), coefficients(1), coefficients(1), coefficients(2);
95  return covariance_matrix;
96 }

109 template <class Range, class Scalar>
110 Eigen::Matrix2<Scalar> calculate_covariance(Range&& range, const Eigen::Vector2<Scalar>& mean) {
111  const auto sample_count = range.size();
112  return calculate_covariance(
113  range,
114  ranges::views::repeat_n(1.0 / static_cast<Scalar>(sample_count), static_cast<std::ptrdiff_t>(sample_count)),
115  mean);

136 template <
137  class Poses,
138  class Weights,
139  class Pose = ranges::range_value_t<Poses>,
140  class Scalar = typename Pose::Scalar,
141  typename = std::enable_if_t<std::is_same_v<Pose, typename Sophus::SE2<Scalar>>>>
142 std::pair<Sophus::SE2<Scalar>, Eigen::Matrix3<Scalar>> estimate(Poses&& poses, Weights&& weights) {
143  auto translation_view = poses | ranges::views::transform([](const auto& pose) { return pose.translation(); });
144  auto weights_view = weights | ranges::views::common;
145  const auto weights_sum = std::accumulate(weights_view.begin(), weights_view.end(), 0.0);
146  auto normalized_weights_view =
147  weights_view | ranges::views::transform([weights_sum](const auto& weight) { return weight / weights_sum; });
148 
149  // map sophus pose 2D pose into a 4D Eigen vector. Mapping exploits that Sophus stores the 2D transform
150  // as two elements for the linear translation, and two more for the orientation (in complex number form)
151  const auto pose_to_weighted_eigen_vector = [](const auto& pose, const auto& weight) {
152  return Eigen::Map<const Eigen::Vector4<Scalar>>{pose.data()} * weight;
153  };
154 
155  // Compute the average of all the coefficients of the SE2 group elements and construct a new SE2 element. Notice
156  // that after averaging the complex representation of the orientation the resulting complex is not on the unit circle.
157  // This is expected and the value will be renormalized after having used the non-normal result to estimate the
158  // orientation autocovariance.
159  const Eigen::Vector4<Scalar> mean_pose_vector = std::transform_reduce(
160  poses.begin(), poses.end(), normalized_weights_view.begin(), Eigen::Vector4<Scalar>::Zero().eval(), std::plus{},
161  pose_to_weighted_eigen_vector);
162 
163  // Calculate the weighted pose estimation
164  Sophus::SE2<Scalar> estimated_pose = Eigen::Map<const Sophus::SE2<Scalar>>{mean_pose_vector.data()};
165 
166  Eigen::Matrix3<Scalar> covariance_matrix = Eigen::Matrix3<Scalar>::Zero();
167 
168  // Compute the covariance of the translation part.
169  covariance_matrix.template topLeftCorner<2, 2>() =
170  calculate_covariance(translation_view, normalized_weights_view, estimated_pose.translation());
171 
172  // Compute the orientation variance and re-normalize the rotation component.
173  if (estimated_pose.so2().unit_complex().norm() < std::numeric_limits<double>::epsilon()) {
174  // Handle the case where both averages are too close to zero.
175  // Return zero yaw and infinite variance.
176  covariance_matrix.coeffRef(2, 2) = std::numeric_limits<double>::infinity();
177  estimated_pose.so2() = Sophus::SO2<Scalar>{0.0};
178  } else {
179  // See circular standard deviation in
180  // https://en.wikipedia.org/wiki/Directional_statistics#Dispersion.
181  covariance_matrix.coeffRef(2, 2) = -2.0 * std::log(estimated_pose.so2().unit_complex().norm());
182  estimated_pose.so2().normalize();
183  }
184  return std::pair{estimated_pose, covariance_matrix};
185 }
