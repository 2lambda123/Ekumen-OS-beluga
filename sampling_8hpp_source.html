189 inline auto kld_condition(std::size_t min, double epsilon, double z = 3.) {
190  auto target_size = [two_epsilon = 2 * epsilon, z](std::size_t k) {
191  if (k <= 2U) {
192  return std::numeric_limits<std::size_t>::max();
193  }
194  double common = 2. / static_cast<double>(9 * (k - 1));
195  double base = 1. - common + std::sqrt(common) * z;
196  double result = (static_cast<double>(k - 1) / two_epsilon) * base * base * base;
197  return static_cast<std::size_t>(std::ceil(result));
198  };
199 
200  return [=, count = 0ULL, buckets = std::unordered_set<std::size_t>{}](std::size_t hash) mutable {
201  count++;
202  buckets.insert(hash);
203  return count <= min || count <= target_size(buckets.size());
204  };
205 }

336  template <class Range>
337  [[nodiscard]] auto generate_samples_from(Range&& particles) {
338  auto&& weights = views::weights(particles);
339  double total_weight = std::reduce(std::begin(weights), std::end(weights), 0.);
340  double average_weight = total_weight / static_cast<double>(weights.size());
341  double random_state_probability = std::max(0., 1. - fast_filter_(average_weight) / slow_filter_(average_weight));
342 
343  if (random_state_probability > 0.) {
344  fast_filter_.reset();
345  slow_filter_.reset();
346  }
347 
348  using particle_type = typename std::decay_t<Range>::value_type;
349  return ranges::views::generate(random_select(
350  [this]() { return this->self().generate_random_state(random_number_generator_); },
351  random_sample(views::states(particles), weights, random_number_generator_), random_number_generator_,
352  random_state_probability)) |
353  ranges::views::transform(make_from_state<particle_type>);
354  }
