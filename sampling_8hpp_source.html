156 template <class Range, class Weights, class RandomNumberGenerator>
157 auto make_multinomial_sampler(const Range& samples, const Weights& weights, RandomNumberGenerator& generator) {
158  auto weights_begin = std::begin(weights);
159  auto weights_end = std::end(weights);
160  using difference_type = decltype(weights_end - weights_begin);
161  return [&generator, first = std::begin(samples),
162  distribution = std::discrete_distribution<difference_type>{weights_begin, weights_end}]() mutable {
163  return *(first + distribution(generator));
164  };
165 }

300  template <class Generator>
301  [[nodiscard]] auto generate_samples_from_particles(Generator& gen) {
302  auto weights = this->self().weights() | ranges::views::common;
303  double total_weight = std::reduce(std::begin(weights), std::end(weights), 0.);
304  double average_weight = total_weight / static_cast<double>(weights.size());
305  double random_state_probability = std::max(0., 1. - fast_filter_(average_weight) / slow_filter_(average_weight));
306 
307  if (random_state_probability > 0.) {
308  fast_filter_.reset();
309  slow_filter_.reset();
310  }
311 
312  auto create_random_state = [this, &gen] { return this->self().make_random_state(gen); };
313  auto sample_existing_state = make_multinomial_sampler(this->self().states(), weights, gen);
314  return ranges::views::generate(make_random_selector(
315  std::move(create_random_state), std::move(sample_existing_state), gen, random_state_probability));
316  }
317 
318  private:
319  ExponentialFilter slow_filter_;
320  ExponentialFilter fast_filter_;
321 };
