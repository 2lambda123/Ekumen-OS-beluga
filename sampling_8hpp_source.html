186 inline auto kld_condition(std::size_t min, double epsilon, double z = 3.) {
187  auto target_size = [two_epsilon = 2 * epsilon, z](std::size_t k) {
188  if (k <= 2U) {
189  return std::numeric_limits<std::size_t>::max();
190  }
191  double common = 2. / static_cast<double>(9 * (k - 1));
192  double base = 1. - common + std::sqrt(common) * z;
193  double result = (static_cast<double>(k - 1) / two_epsilon) * base * base * base;
194  return static_cast<std::size_t>(std::ceil(result));
195  };
196 
197  return [=, count = 0ULL, buckets = std::unordered_set<std::size_t>{}](std::size_t hash) mutable {
198  count++;
199  buckets.insert(hash);
200  return count <= min || count <= target_size(buckets.size());
201  };
202 }

332  template <class Range>
333  [[nodiscard]] auto generate_samples_from(Range&& particles) {
334  auto&& weights = views::weights(particles);
335  double total_weight = std::reduce(std::begin(weights), std::end(weights), 0.);
336  double average_weight = total_weight / static_cast<double>(weights.size());
337  double random_state_probability = std::max(0., 1. - fast_filter_(average_weight) / slow_filter_(average_weight));
338 
339  if (random_state_probability > 0.) {
340  fast_filter_.reset();
341  slow_filter_.reset();
342  }
343 
344  using particle_type = typename std::decay_t<Range>::value_type;
345  return ranges::views::generate(random_select(
346  [this]() { return this->self().generate_random_state(random_number_generator_); },
347  random_sample(views::states(particles), weights, random_number_generator_), random_number_generator_,
348  random_state_probability)) |
349  ranges::views::transform(make_from_state<particle_type>);
350  }
