155 template <class Range, class Weights, class RandomNumberGenerator>
156 auto make_multinomial_sampler(const Range& samples, const Weights& weights, RandomNumberGenerator& generator) {
157  auto weights_begin = std::begin(weights);
158  auto weights_end = std::end(weights);
159  using difference_type = decltype(weights_end - weights_begin);
160  return [&generator, first = std::begin(samples),
161  distribution = std::discrete_distribution<difference_type>{weights_begin, weights_end}]() mutable {
162  return *(first + distribution(generator));
163  };
164 }

217 inline auto kld_condition(std::size_t min, double epsilon, double z = 3.) {
218  auto target_size = [two_epsilon = 2 * epsilon, z](std::size_t k) {
219  if (k <= 2U) {
220  return std::numeric_limits<std::size_t>::max();
221  }
222  double common = 2. / static_cast<double>(9 * (k - 1));
223  double base = 1. - common + std::sqrt(common) * z;
224  double result = (static_cast<double>(k - 1) / two_epsilon) * base * base * base;
225  return static_cast<std::size_t>(std::ceil(result));
226  };
227 
228  return [=, count = 0ULL, buckets = std::unordered_set<std::size_t>{}](std::size_t hash) mutable {
229  count++;
230  buckets.insert(hash);
231  return count <= min || count <= target_size(buckets.size());
232  };

368  template <class Generator>
369  [[nodiscard]] auto generate_samples_from_particles(Generator& gen) {
370  auto weights = this->self().weights() | ranges::views::common;
371  double total_weight = std::reduce(std::begin(weights), std::end(weights), 0.);
372  double average_weight = total_weight / static_cast<double>(weights.size());
373  double random_state_probability = std::max(0., 1. - fast_filter_(average_weight) / slow_filter_(average_weight));
374 
375  if (random_state_probability > 0.) {
376  fast_filter_.reset();
377  slow_filter_.reset();
378  }
379 
380  auto create_random_state = [this, &gen] { return this->self().make_random_state(gen); };
381  auto sample_existing_state = make_multinomial_sampler(this->self().states(), weights, gen);
382  return ranges::views::generate(make_random_selector(
383  std::move(create_random_state), std::move(sample_existing_state), gen, random_state_probability));
384  }
385 
386  private:
387  ExponentialFilter slow_filter_;
388  ExponentialFilter fast_filter_;
