135  template <typename I, typename S>
136  constexpr void assign(I first, S last) {
137  auto range = ranges::make_subrange(first, last);
138  static_assert(ranges::input_range<decltype(range)>);
139  if constexpr (ranges::sized_range<decltype(range)>) {
140  resize(ranges::size(range));
141  ranges::copy(range, begin());
142  } else {
143  // Optimization to copy as much as we can before we start back inserting.
144  // Back insertion time is severely punished by having multiple containers.
145  //
146  // There are two bad cases for this implementation:
147  // - The capacity is too large compared to the size of the input range, so
148  // we allocate and default construct unnecessarily.
149  // - The capacity is too low compared to the size of the input range, so
150  // there will be a lot of back inserting.
151  //
152  // If common guidelines of calling reserve with the expected size are followed,
153  // then this is quite fast, even when we don't know the exact size of the range
154  // in advance.
155  resize(capacity());
156  // Copy elements until we reach current capacity.
157  auto limited_range = range | ranges::views::take(size());
158  auto [last_in, last_out] = ranges::copy(limited_range, begin());
159  const auto copied_size = static_cast<size_type>(ranges::distance(begin(), last_out));
160  if (size() == copied_size) {
161  // Back insert the remaining elements if any.
162  ranges::copy(last_in.base(), last, ranges::back_inserter(*this));
163  } else {
164  // Remove extra elements by resizing to the correct size.
165  resize(copied_size);
166  }
167  }
168  }
