131  [[nodiscard]] weight_type importance_weight(const state_type& state) const {
132  const auto transform = grid_.origin().inverse() * state;
133  // TODO(glpuga): Investigate why AMCL and QuickMCL both use this formula for the weight.
134  // See https://github.com/Ekumen-OS/beluga/issues/153
135  return std::transform_reduce(
136  points_.cbegin(), points_.cend(), 1.0, std::plus{},
137  [this, x_offset = transform.translation().x(), y_offset = transform.translation().y(),
138  cos_theta = transform.so2().unit_complex().x(), sin_theta = transform.so2().unit_complex().y(),
139  unknown_space_occupancy_prob = 1. / params_.max_laser_distance](const auto& point) {
140  // Transform the end point of the laser to the grid local coordinate system.
141  // Not using Eigen/Sophus because they make the routine x10 slower.
142  // See `benchmark_likelihood_field_model.cpp` for reference.
143  const auto x = point.first * cos_theta - point.second * sin_theta + x_offset;
144  const auto y = point.first * sin_theta + point.second * cos_theta + y_offset;
145  const auto pz = likelihood_field_.data_near(x, y).value_or(unknown_space_occupancy_prob);
146  return pz * pz * pz;
147  });
148  }

172  static ValueGrid2<double> make_likelihood_field(const LikelihoodFieldModelParam& params, const OccupancyGrid& grid) {
173  const auto squared_distance = [&grid,
174  squared_max_distance = params.max_obstacle_distance * params.max_obstacle_distance](
175  std::size_t first, std::size_t second) {
176  return std::min((grid.coordinates_at(first) - grid.coordinates_at(second)).squaredNorm(), squared_max_distance);
177  };
178 
179  const auto neighborhood = [&grid](std::size_t index) { return grid.neighborhood4(index); };
180 
181  const auto distance_map = nearest_obstacle_distance_map(grid.obstacle_data(), squared_distance, neighborhood);
182 
183  const auto to_likelihood = [amplitude =
184  params.z_hit / (params.sigma_hit * std::sqrt(2 * Sophus::Constants<double>::pi())),
185  two_squared_sigma = 2 * params.sigma_hit * params.sigma_hit,
186  offset = params.z_random / params.max_laser_distance](double squared_distance) {
187  return amplitude * std::exp(-squared_distance / two_squared_sigma) + offset;
188  };
189 
190  auto likelihood_data = distance_map | ranges::views::transform(to_likelihood) | ranges::to<std::vector>;
191  return ValueGrid2<double>{std::move(likelihood_data), grid.width(), grid.resolution()};
192  }
