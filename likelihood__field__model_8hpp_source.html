155  [[nodiscard]] weight_type importance_weight(const state_type& state) const {
156  const auto transform = grid_.origin().inverse() * state;
157  const auto lock = std::shared_lock<std::shared_mutex>{points_mutex_};
158  return std::transform_reduce(
159  points_.cbegin(), points_.cend(), 0.0, std::plus{},
160  [this, x_offset = transform.translation().x(), y_offset = transform.translation().y(),
161  cos_theta = transform.so2().unit_complex().x(),
162  sin_theta = transform.so2().unit_complex().y()](const auto& point) {
163  // Transform the end point of the laser to the global coordinate system.
164  // Not using Eigen/Sophus because they make the routine x10 slower.
165  // See `benchmark_likelihood_field_model.cpp` for reference.
166  const auto x = point.first * cos_theta - point.second * sin_theta + x_offset;
167  const auto y = point.first * sin_theta + point.second * cos_theta + y_offset;
168  const auto index = grid_.index(x, y);
169  return index < likelihood_field_.size() ? likelihood_field_[index] : 0.0;
170  });
171  }

186  static std::vector<std::size_t> make_free_cells_vector(const OccupancyGrid& grid) {
187  auto free_cells = std::vector<std::size_t>{};
188  free_cells.reserve(grid.size());

197  static std::vector<double> make_likelihood_field(const LikelihoodFieldModelParam& params, const OccupancyGrid& grid) {
198  const auto obstacle_map = grid.data() | ranges::views::transform(&OccupancyGrid::Traits::is_occupied);
199 
200  auto squared_distance = [&grid, squared_max_distance = params.max_obstacle_distance * params.max_obstacle_distance](
201  std::size_t first, std::size_t second) {
202  return std::min((grid.point(first) - grid.point(second)).squaredNorm(), squared_max_distance);
203  };
204 
205  const auto distance_map = nearest_obstacle_distance_map(
206  obstacle_map, squared_distance, std::bind(&OccupancyGrid::neighbors, &grid, std::placeholders::_1));
207 
208  auto to_likelihood = [amplitude =
209  params.z_hit / (params.sigma_hit * std::sqrt(2 * Sophus::Constants<double>::pi())),
210  two_squared_sigma = 2 * params.sigma_hit * params.sigma_hit,
211  offset = params.z_random / params.max_laser_distance](double squared_distance) {
212  return amplitude * std::exp(-squared_distance / two_squared_sigma) + offset;
213  };
214 
215  return distance_map | ranges::views::transform(to_likelihood) | ranges::to<std::vector>;
216  }
