104  template <class Generator>
105  [[nodiscard]] state_type apply_motion(const state_type& state, Generator& gen) const {
106  static thread_local auto distribution = std::normal_distribution<double>{};
107  const auto lock = std::shared_lock<std::shared_mutex>{params_mutex_};
108  const auto first_rotation = Sophus::SO2d{distribution(gen, first_rotation_params_)};
109  const auto translation = Eigen::Vector2d{distribution(gen, translation_params_), 0.0};
110  const auto second_rotation = Sophus::SO2d{distribution(gen, second_rotation_params_)};
111  return state * Sophus::SE2d{first_rotation, Eigen::Vector2d{0.0, 0.0}} * Sophus::SE2d{second_rotation, translation};
112  }

115  void update_motion(const update_type& pose) final {
116  if (last_pose_) {
117  const auto translation = pose.translation() - last_pose_.value().translation();
118  const double distance = translation.norm();
119  const double distance_variance = distance * distance;
120 
121  const auto& previous_orientation = last_pose_.value().so2();
122  const auto& current_orientation = pose.so2();
123  const auto first_rotation =
124  distance > params_.distance_threshold
125  ? Sophus::SO2d{std::atan2(translation.y(), translation.x())} * previous_orientation.inverse()
126  : Sophus::SO2d{0.0};
127  const auto second_rotation = current_orientation * previous_orientation.inverse() * first_rotation.inverse();
128  const auto combined_rotation = first_rotation * second_rotation;
129 
130  {
131  const auto lock = std::lock_guard<std::shared_mutex>{params_mutex_};
132  first_rotation_params_ = DistributionParam{
133  first_rotation.log(), std::sqrt(
134  params_.rotation_noise_from_rotation * rotation_variance(first_rotation) +
135  params_.rotation_noise_from_translation * distance_variance)};
136  translation_params_ = DistributionParam{
137  distance, std::sqrt(
138  params_.translation_noise_from_translation * distance_variance +
139  params_.translation_noise_from_rotation * rotation_variance(combined_rotation))};
140  second_rotation_params_ = DistributionParam{
141  second_rotation.log(), std::sqrt(
142  params_.rotation_noise_from_rotation * rotation_variance(second_rotation) +
143  params_.rotation_noise_from_translation * distance_variance)};
144  }
145  }
146  last_pose_ = pose;
147  }

166  static double rotation_variance(const Sophus::SO2d& rotation) {
167  // Treat backward and forward motion symmetrically for the noise models.
168  const auto flipped_rotation = rotation * Sophus::SO2d{Sophus::Constants<double>::pi()};
169  const auto delta = std::min(std::abs(rotation.log()), std::abs(flipped_rotation.log()));
170  return delta * delta;
171  }
