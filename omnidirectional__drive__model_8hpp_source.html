102  [[nodiscard]] state_type apply_motion(const state_type& state) const {
103  static thread_local auto generator = std::mt19937{std::random_device()()};
104  static thread_local auto distribution = std::normal_distribution<double>{};
105  const auto lock = std::shared_lock<std::shared_mutex>{params_mutex_};
106  // This is an implementation based on the same set of parameters that is used in
107  // nav2's omni_motion_model. To simplify the implementation, the following
108  // variable substitutions were performed:
109  // - first_rotation = delta_bearing - previous_orientation
110  // - second_rotation = delta_rot_hat - first_rotation
111  const auto second_rotation = Sophus::SO2d{distribution(generator, rotation_params_)} * first_rotation_.inverse();
112  const auto translation = Eigen::Vector2d{
113  distribution(generator, translation_params_),
114  -1.0 * distribution(generator, strafe_params_),
115  };
116  return state * Sophus::SE2d{first_rotation_, Eigen::Vector2d{0.0, 0.0}} *
117  Sophus::SE2d{second_rotation, translation};
118  }

128  void update_motion(const update_type& pose) {
129  if (last_pose_) {
130  const auto translation = pose.translation() - last_pose_.value().translation();
131  const double distance = translation.norm();
132  const double distance_variance = distance * distance;
133 
134  const auto& previous_orientation = last_pose_.value().so2();
135  const auto& current_orientation = pose.so2();
136  const auto rotation = current_orientation * previous_orientation.inverse();
137 
138  {
139  const auto lock = std::lock_guard<std::shared_mutex>{params_mutex_};
140  first_rotation_ =
141  distance > params_.distance_threshold
142  ? Sophus::SO2d{std::atan2(translation.y(), translation.x())} * previous_orientation.inverse()
143  : Sophus::SO2d{0.0};
144  const double rotation_variance_ = rotation_variance(rotation);
145  rotation_params_ = DistributionParam{
146  rotation.log(), std::sqrt(
147  params_.rotation_noise_from_rotation * rotation_variance_ +
148  params_.rotation_noise_from_translation * distance_variance)};
149  translation_params_ = DistributionParam{
150  distance, std::sqrt(
151  params_.translation_noise_from_translation * distance_variance +
152  params_.translation_noise_from_rotation * rotation_variance_)};
153  strafe_params_ = DistributionParam{
154  0.0, std::sqrt(
155  params_.translation_strafe_noise_from_translation * distance_variance +
156  params_.translation_noise_from_rotation * rotation_variance_)};
157  }
158  }
159  last_pose_ = pose;
160  }

174  static double rotation_variance(const Sophus::SO2d& rotation) {
175  // Treat backward and forward motion symmetrically for the noise models.
176  const auto flipped_rotation = rotation * Sophus::SO2d{Sophus::Constants<double>::pi()};
177  const auto delta = std::min(std::abs(rotation.log()), std::abs(flipped_rotation.log()));
178  return delta * delta;
179  }
