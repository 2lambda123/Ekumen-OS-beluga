67 template <class T, class... Args>
68 constexpr std::size_t tuple_index_helper() noexcept {
69  constexpr bool kMatches[sizeof...(Args)] = // NOLINT(modernize-avoid-c-arrays)
70  {std::is_same<T, std::decay_t<Args>>::value...};
71  std::size_t selected = kTupleIndexNotFound;
72  for (std::size_t i = 0; i < sizeof...(Args); ++i) {
73  if (kMatches[i]) {
74  if (selected == kTupleIndexNotFound) {
75  selected = i;
76  } else {
77  return kTupleIndexAmbiguous;
78  }
79  }
80  }
