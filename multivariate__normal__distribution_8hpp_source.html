103  [[nodiscard]] static Matrix make_transform(const Matrix& covariance) {
104  // For more information about the method used to generate correlated normal vectors
105  // from independent normal distributions, see:
106  // https://en.wikipedia.org/wiki/Multivariate_normal_distribution#Drawing_values_from_the_distribution
107  // Gentle, J. E. (2009). Computational Statistics. Statistics and Computing. New York: Springer. pp. 315–316.
108  // https://doi.org/10.1007%2F978-0-387-98144-4
109  if (!covariance.isApprox(covariance.transpose())) {
110  throw std::runtime_error("Invalid covariance matrix, it is not symmetric.");
111  }
112  const auto solver = Eigen::SelfAdjointEigenSolver<Matrix>{covariance};
113  if (solver.info() != Eigen::Success) {
114  throw std::runtime_error("Invalid covariance matrix, eigen solver failed.");
115  }
116  const auto& eigenvalues = solver.eigenvalues();
117  if ((eigenvalues.array() < 0.0).any()) {
118  throw std::runtime_error("Invalid covariance matrix, it has negative eigenvalues.");
119  }
120  return solver.eigenvectors() * eigenvalues.cwiseSqrt().asDiagonal();
121  }
122  };
