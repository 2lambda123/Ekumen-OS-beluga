90  [[nodiscard]] state_type make_random_state(Generator& gen) const {
91  const auto map_boundaries = landmark_map_.map_limits();
92 
93  if constexpr (std::is_same_v<state_type, Sophus::SE3d>) {
94  auto x_distribution = std::uniform_real_distribution<double>{map_boundaries.x_min, map_boundaries.x_max};
95  auto y_distribution = std::uniform_real_distribution<double>{map_boundaries.y_min, map_boundaries.y_max};
96  auto z_distribution = std::uniform_real_distribution<double>{map_boundaries.z_min, map_boundaries.z_max};
97  return Sophus::SE3d{
98  Sophus::SO3d::sampleUniform(gen),
99  Eigen::Vector3d{x_distribution(gen), y_distribution(gen), z_distribution(gen)}};
100  } else {
101  auto x_distribution = std::uniform_real_distribution<double>{map_boundaries.x_min, map_boundaries.x_max};
102  auto y_distribution = std::uniform_real_distribution<double>{map_boundaries.y_min, map_boundaries.y_max};
103  return Sophus::SE2d{Sophus::SO2d::sampleUniform(gen), Eigen::Vector2d{x_distribution(gen), y_distribution(gen)}};

112  [[nodiscard]] weight_type importance_weight(const state_type& state) const {
113  Sophus::SE3d robot_pose_in_world;
114 
115  if constexpr (std::is_same_v<state_type, Sophus::SE3d>) {
116  // The robot pose state is already given in 3D,
117  robot_pose_in_world = state;
118  } else {
119  // The robot pose state is given in 2D. Notice that in this case
120  // the 2D pose of the robot is assumed to be that of the robot footprint (projection of the robot
121  // on the z=0 plane of the 3D world frame). This is so that we can tie the sensor reference frame
122  // to the world frame where the landmarks are given without additional structural information.
123  robot_pose_in_world = Sophus::SE3d{
124  Sophus::SO3d::rotZ(state.so2().log()),
125  Eigen::Vector3d{state.translation().x(), state.translation().y(), 0.0}};
126  }
127 
128  // precalculate the sensor pose in the world frame
129  const auto sensor_pose_in_world = robot_pose_in_world * params_.sensor_pose_in_robot;
130 
131  const auto detection_weight = [this, &sensor_pose_in_world](const auto& detection) {
132  // find the landmark the most closely matches the sample bearing vector
133  const auto opt_landmark_bearing_in_sensor = landmark_map_.find_closest_bearing_landmark(
134  detection.detection_bearing_in_sensor, detection.category, sensor_pose_in_world);
135 
136  // if we did not find a matching landmark, return 0.0
137  if (!opt_landmark_bearing_in_sensor) {
138  return 0.0;
139  }
140 
141  // recover sample bearing vector
142  const auto detection_bearing_in_sensor = detection.detection_bearing_in_sensor.normalized();
143 
144  // recover landmark bearing vector
145  const auto& landmark_bearing_in_sensor = *opt_landmark_bearing_in_sensor;
146 
147  // calculate the aperture angle between the detection and the landmark
148  const auto cos_aperture = detection_bearing_in_sensor.dot(landmark_bearing_in_sensor);
149  const auto sin_aperture = detection_bearing_in_sensor.cross(landmark_bearing_in_sensor).norm();
150 
151  // calculate the angle between the sample and the landmark
152  const auto bearing_error = std::atan2(sin_aperture, cos_aperture);
153 
154  // model the probability of the landmark being detected as depending on the bearing error
155  const auto bearing_error_prob =
156  std::exp(-bearing_error * bearing_error / (2. * params_.sigma_bearing * params_.sigma_bearing));
157 
158  // We'll assume the probability of identification error to be zero
159  const auto prob = bearing_error_prob;
160 
161  // TODO(unknown): We continue to use the sum-of-cubes formula that nav2 uses
162  // See https://github.com/Ekumen-OS/beluga/issues/153
163  return prob * prob * prob;
164  };
165 
166  return std::transform_reduce(points_.cbegin(), points_.cend(), 1.0, std::plus{}, detection_weight);
167  }
