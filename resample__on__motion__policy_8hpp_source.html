59  template <typename Concrete>
60  [[nodiscard]] bool do_resampling(Concrete& filter) {
61  // To avoid loss of diversity in the particle population, don't
62  // resample when the state is known to be static.
63  // Probabilistic Robotics \cite thrun2005probabilistic Chapter 4.2.4.
64  auto current_pose = filter.latest_motion_update();
65 
66  // default to letting other policies decide
67  bool must_do_resample{true};
68 
69  if (current_pose && latest_resample_pose_) {
70  // calculate relative transform between previous pose and the current one
71  const auto delta = latest_resample_pose_->inverse() * current_pose.value();
72 
73  // only resample if movement is above thresholds
74  must_do_resample = //
75  std::abs(delta.translation().x()) > configuration_.update_min_d ||
76  std::abs(delta.translation().y()) > configuration_.update_min_d ||
77  std::abs(delta.so2().log()) > configuration_.update_min_a;
78  }
79 
80  // we always measure the distance traveled since the last time we did resample
81  if (must_do_resample) {
82  latest_resample_pose_ = current_pose;
83  }
84 
85  return must_do_resample;
86  }
87 
88  private:
89  param_type configuration_;
90  std::optional<motion_event> latest_resample_pose_;
91 };
