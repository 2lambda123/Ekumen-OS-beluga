201  template <typename ExecutionPolicy>
202  void sample_impl(ExecutionPolicy&& policy) {
203  auto states = this->self().states() | ranges::views::common;
204  std::transform(
205  std::forward<ExecutionPolicy>(policy), std::begin(states), std::end(states), std::begin(states),

209  template <typename ExecutionPolicy>
210  void reweight_impl(ExecutionPolicy&& policy) {
211  auto states = this->self().states() | ranges::views::common;
212  auto weights = this->self().weights() | ranges::views::common;
213  std::transform(
214  std::forward<ExecutionPolicy>(policy), std::begin(states), std::end(states), std::begin(weights),
215  std::begin(weights),
216  [this](const auto& state, auto weight) { return weight * this->self().importance_weight(state); });
217 
218  // max normalize weights to avoid unbounded growth
219  const auto weight_maximum_norm = *ranges::max_element(weights);
220  std::transform(std::begin(weights), std::end(weights), std::begin(weights), [weight_maximum_norm](auto weight) {
221  return weight / weight_maximum_norm;
222  });
