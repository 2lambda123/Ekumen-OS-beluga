74  [[nodiscard]] auto trace(const Sophus::SO2d& bearing) const {
75  const auto far_end_pose_in_source_frame = Sophus::SE2d{
76  Sophus::SO2d{0.},
77  Eigen::Vector2d{max_range_ * bearing.unit_complex().x(), max_range_ * bearing.unit_complex().y()}};
78  const auto far_end_pose_in_grid_frame = source_pose_in_grid_frame_ * far_end_pose_in_source_frame;
79  const auto start_cell = grid_.cell(source_pose_in_grid_frame_.translation());
80  const auto end_cell = grid_.cell(far_end_pose_in_grid_frame.translation());
81  const auto cell_is_valid = [this](const auto& cell) { return grid_.valid(cell); };
82  return algorithm_(start_cell, end_cell) | ranges::views::take_while(cell_is_valid);
83  }

92  [[nodiscard]] std::optional<double> cast(const Sophus::SO2d& bearing) const {
93  const auto is_free = [this](const auto& cell) {
94  // TODO(hidmic): move to occupancy grid API
95  return Grid::Traits::is_free(grid_.data()[grid_.index(cell)]);
96  };
97  for (const auto& cell : trace(bearing)) {
98  if (!is_free(cell)) {
99  const auto start_cell = grid_.cell(source_pose_in_grid_frame_.translation());
100  const auto distance = (grid_.point(cell) - grid_.point(start_cell)).norm();
101  return std::make_optional(std::min(distance, max_range_));
102  }
